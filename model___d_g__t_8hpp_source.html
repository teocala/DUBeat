192 template <class basis>
193 void
194 ModelDG_t<basis>::parse_parameters(lifex::ParamHandler &params)
195 {
196  // Parse input file.
197  params.parse();
198  // Read input parameters.
199  this->linear_solver.parse_parameters(params);
200  this->preconditioner.parse_parameters(params);
201 
202  // Extra parameters.
203  params.enter_subsection("Mesh and space discretization");
204  this->prm_n_refinements = params.get_integer("Number of refinements");
205  this->prm_fe_degree = params.get_integer("FE space degree");
206  params.leave_subsection();
207 
208  params.enter_subsection("Discontinuous Galerkin");
209  this->prm_penalty_coeff = params.get_double("Penalty coefficient");
210  AssertThrow(this->prm_penalty_coeff == 1. || this->prm_penalty_coeff == 0. ||
211  this->prm_penalty_coeff == -1.,
212  dealii::StandardExceptions::ExcMessage(
213  "Penalty coefficient must be 1 (SIP method) or 0 (IIP method) "
214  "or -1 (NIP method)."));
215 
216  this->prm_stability_coeff = params.get_double("Stability coefficient");
217  params.leave_subsection();
218 
219  params.enter_subsection("Time solver");
220  this->prm_time_init = params.get_double("Initial time");
221  this->prm_time_final = params.get_double("Final time");
222  AssertThrow(this->prm_time_final > this->prm_time_init,
223  dealii::StandardExceptions::ExcMessage(
224  "Final time must be greater than initial time."));
225 
226  this->prm_time_step = params.get_double("Time step");
227  this->prm_bdf_order = params.get_integer("BDF order");
228  params.leave_subsection();
229 }

231 template <class basis>
232 void
233 ModelDG_t<basis>::time_initialization()
234 {
235  // Set initial time to the exact analytical solution.
236  this->u_ex->set_time(prm_time_init);
237 
238  // Solution_owned and solution_ex_owned at the initial time are the
239  // discretization of the analytical u_ex.
240  this->discretize_analytical_solution(this->u_ex, this->solution_owned);
241  this->solution_ex_owned = this->solution_owned;
242  this->solution = this->solution_owned;
243 
244  // Initialization of the initial solution.
245  const std::vector<lifex::LinAlg::MPI::Vector> sol_init(this->prm_bdf_order,
246  this->solution_owned);
247 
248  // Initialization of the BDFHandler
249  bdf_handler.initialize(this->prm_bdf_order, sol_init);
250 }

252 template <class basis>
253 void
254 ModelDG_t<basis>::intermediate_error_print(
255  const lifex::LinAlg::MPI::Vector &solution_owned,
256  const lifex::LinAlg::MPI::Vector &solution_ex_owned,
257  const std::shared_ptr<dealii::Function<lifex::dim>> &u_ex,
258  const char *solution_name)
259 {
260  AssertThrow(u_ex != nullptr,
261  dealii::StandardExceptions::ExcMessage(
262  "Not valid pointer to the exact solution."));
263 
264  AssertThrow(solution_owned.size() == solution_ex_owned.size(),
265  dealii::StandardExceptions::ExcMessage(
266  "The exact solution vector and the approximate solution vector "
267  "must have the same length."));
268 
269  lifex::LinAlg::MPI::Vector error_owned =
270  this->conversion_to_fem(solution_owned);
271  error_owned -= this->conversion_to_fem(solution_ex_owned);
272 
273  pcerr << solution_name << ":"
274  << "\tL-inf error norm: " << error_owned.linfty_norm() << std::endl;
275 }

279 ModelDG_t<basis>::update_time()
280 {
281  // Update time for all the known analytical functions.
282  this->u_ex->set_time(this->time);
283  this->f_ex->set_time(this->time);
284  this->g_n->set_time(this->time);
285  this->grad_u_ex->set_time(this->time);
286 
287  bdf_handler.time_advance(this->solution_owned, true);
288  this->solution_bdf = bdf_handler.get_sol_bdf();
289 
290  // Update solution_ex_owned from the updated u_ex.
291  this->discretize_analytical_solution(this->u_ex, this->solution_ex_owned);
292 }

294 template <class basis>
295 void
296 ModelDG_t<basis>::run()
297 {
298  this->create_mesh();
299  this->setup_system();
300  this->initialize_solution(this->solution_owned, this->solution);
301  this->initialize_solution(this->solution_ex_owned, this->solution_ex);
302  this->time_initialization();
303 
304  while (this->time < this->prm_time_final)
305  {
306  time += prm_time_step;
307  ++timestep_number;
308 
309  pcout << "Time step " << std::setw(6) << timestep_number
310  << " at t = " << std::setw(8) << std::fixed << std::setprecision(6)
311  << time << std::endl;
312 
313  this->update_time();
314  this->solution_ext = bdf_handler.get_sol_extrapolation();
315 
316  this->assemble_system();
317 
318  // Initial guess.
319  this->solution_owned = this->solution_ext;
320  this->solve_system();
321 
322  this->intermediate_error_print(this->solution_owned,
323  this->solution_ex_owned,
324  this->u_ex);
325  }
326 
327  this->compute_errors(this->solution_owned,
328  this->solution_ex_owned,
329  this->u_ex,
330  this->grad_u_ex);
331 
332 
333  // Generation of the graphical output.
334  this->solution_ex = this->solution_ex_owned;
335  this->conversion_to_fem(this->solution_ex);
336  this->solution = this->solution_owned;
337  this->conversion_to_fem(this->solution);
338  this->output_results();
339 }
