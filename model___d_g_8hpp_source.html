307 template <class basis>
308 unsigned int
309 ModelDG<basis>::get_dofs_per_cell() const
310 {
311  // The analytical formula is:
312  // n_dof_per_cell = (p+1)*(p+2)*...(p+d) / d!,
313  // where p is the space order and d the space dimension..
314 
315  unsigned int denominator = 1;
316  unsigned int nominator = 1;
317 
318  for (unsigned int i = 1; i <= lifex::dim; i++)
319  {
320  denominator *= i;
321  nominator *= prm_fe_degree + i;
322  }
323 
324  return (int)(nominator / denominator);
325 }

328 template <class basis>
329 void
330 ModelDG<basis>::run()
331 {
332  // Initialization
333  create_mesh();
334  setup_system();
335  initialize_solution(solution_owned, solution);
336  initialize_solution(solution_ex_owned, solution_ex);
337  discretize_analytical_solution(u_ex, solution_ex_owned);
338 
339 
340  // Initial guess.
341  solution_ex = solution_ex_owned;
342  solution = solution_owned = 0;
343 
344  // Computation of the numerical solution.
345  assemble_system();
346  solve_system();
347 
348 
349  // Computation and output of the errors.
350  compute_errors(solution_owned, solution_ex_owned, u_ex, grad_u_ex, "u");
351 
352  // Generation of the graphical output.
353  conversion_to_fem(solution_ex);
354  solution = solution_owned;
355  conversion_to_fem(solution);
356  output_results();
357 }

359 template <class basis>
360 void
361 ModelDG<basis>::setup_system()
362 {
363  assemble = std::make_unique<AssembleDG<basis>>(prm_fe_degree);
364 
365  dof_handler.reinit(triangulation->get());
366  dof_handler.distribute_dofs(prm_fe_degree);
367  dub_fem_values =
368  std::make_shared<DUBFEMHandler<basis>>(prm_fe_degree, dof_handler);
369 
370  triangulation->get_info().print(prm_subsection_path,
371  dof_handler.n_dofs(),
372  true);
373 
374  dealii::IndexSet owned_dofs = dof_handler.locally_owned_dofs();
375  dealii::IndexSet relevant_dofs = owned_dofs;
376  dealii::DynamicSparsityPattern dsp(relevant_dofs);
377 
378 
379  // Add (dof, dof_neigh) to dsp, so to the matrix
380  dofs_per_cell = get_dofs_per_cell();
381  std::vector<lifex::types::global_dof_index> dof_indices(dofs_per_cell);
382  std::vector<lifex::types::global_dof_index> dof_indices_neigh(dofs_per_cell);
383 
384 
385  for (const auto &cell : dof_handler.active_cell_iterators())
386  {
387  dof_indices = dof_handler.get_dof_indices(cell);
388 
389  for (const auto &edge : cell->face_indices())
390  {
391  if (!cell->at_boundary(edge))
392  {
393  const auto neighcell = cell->neighbor(edge);
394  dof_indices_neigh = dof_handler.get_dof_indices(neighcell);
395  for (unsigned int i = 0; i < dofs_per_cell; ++i)
396  {
397  for (unsigned int j = 0; j < dofs_per_cell; ++j)
398  {
399  dsp.add(dof_indices[i], dof_indices_neigh[j]);
400  }
401  }
402  }
403  }
404  }
405 
406  make_sparsity_pattern(dof_handler, dsp);
407  lifex::SparsityTools::distribute_sparsity_pattern(dsp,
408  owned_dofs,
409  mpi_comm,
410  relevant_dofs);
411  lifex::utils::initialize_matrix(matrix, owned_dofs, dsp);
412 
413  rhs.reinit(owned_dofs, mpi_comm);
414 }

417 template <class basis>
418 void
419 ModelDG<basis>::make_sparsity_pattern(
420  const DoFHandlerDG<basis> &dof,
421  dealii::DynamicSparsityPattern &sparsity,
422  const dealii::AffineConstraints<double> &constraints,
423  const bool keep_constrained_dofs,
424  const dealii::types::subdomain_id subdomain_id)
425 {
426  Assert(sparsity.n_rows() == n_dofs,
427  ExcDimensionMismatch(sparsity.n_rows(), n_dofs));
428  Assert(sparsity.n_cols() == n_dofs,
429  ExcDimensionMismatch(sparsity.n_cols(), n_dofs));
430 
431  // If we have a distributed Triangulation only allow locally_owned
432  // subdomain. Not setting a subdomain is also okay, because we skip
433  // ghost cells in the loop below.
434  if (const auto *triangulation = dynamic_cast<
435  const dealii::parallel::DistributedTriangulationBase<lifex::dim> *>(
436  &dof.get_triangulation()))
437  {
438  Assert((subdomain_id == numbers::invalid_subdomain_id) ||
439  (subdomain_id == triangulation->locally_owned_subdomain()),
440  ExcMessage(
441  "For distributed Triangulation objects and associated "
442  "DoFHandler objects, asking for any subdomain other than the "
443  "locally owned one does not make sense."));
444  }
445  std::vector<dealii::types::global_dof_index> dofs_on_this_cell;
446  dofs_on_this_cell.reserve(dof.n_dofs_per_cell());
447 
448  // In case we work with a distributed sparsity pattern of Trilinos
449  // type, we only have to do the work if the current cell is owned by
450  // the calling processor. Otherwise, just continue.
451  for (const auto &cell : dof.active_cell_iterators())
452  if (((subdomain_id == dealii::numbers::invalid_subdomain_id) ||
453  (subdomain_id == cell->subdomain_id())) &&
454  cell->is_locally_owned())
455  {
456  dofs_on_this_cell.resize(dof.n_dofs_per_cell());
457  dofs_on_this_cell = dof.get_dof_indices(cell);
458 
459  // make sparsity pattern for this cell. if no constraints pattern
460  // was given, then the following call acts as if simply no
461  // constraints existed
462  constraints.add_entries_local_to_global(dofs_on_this_cell,
463  sparsity,
464  keep_constrained_dofs);
465  }
466 }

561 {
562  lifex::DataOut<lifex::dim> data_out;

564  // To output results, we need the deal.II DoFHandler instead of the DUBeat
