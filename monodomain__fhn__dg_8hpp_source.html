85  class RightHandSide : public lifex::Function<lifex::dim>
86  {
87  private:
89  double ChiM;
90 
92  double Sigma;
93 
95  double Cm;
96 
98  double kappa;
99 
101  double epsilon;
102 
104  double gamma;
105 
107  double a;
108 
109  public:
111  RightHandSide(double ChiM,
112  double Sigma,
113  double Cm,
114  double kappa,
115  double epsilon,
116  double gamma,
117  double a)
118  : lifex::Function<lifex::dim>()
119  , ChiM(ChiM)
120  , Sigma(Sigma)
121  , Cm(Cm)
122  , kappa(kappa)
123  , epsilon(epsilon)
124  , gamma(gamma)
125  , a(a)
126  {}
127 
129  virtual double
130  value(const dealii::Point<lifex::dim> &p,
131  const unsigned int /*component*/ = 0) const override
132  {
133  if (lifex::dim == 2)
134  return std::sin(2 * M_PI * p[0]) * std::sin(2 * M_PI * p[1]) *
135  std::exp(-5 * this->get_time()) *
136  (-ChiM * Cm * 5 + Sigma * 8 * pow(M_PI, 2) +
137  ChiM * kappa *
138  (std::sin(2 * M_PI * p[0]) * std::sin(2 * M_PI * p[1]) *
139  std::exp(-5 * this->get_time()) -
140  a) *
141  (std::sin(2 * M_PI * p[0]) * std::sin(2 * M_PI * p[1]) *
142  std::exp(-5 * this->get_time()) -
143  1) +
144  ChiM * (epsilon / (epsilon * gamma - 5)));
145  else
146  return std::sin(2 * M_PI * p[0] + M_PI / 4) *
147  std::sin(2 * M_PI * p[1] + M_PI / 4) *
148  std::sin(2 * M_PI * p[2] + M_PI / 4) *
149  std::exp(-5 * this->get_time()) *
150  (-ChiM * Cm * 5 + Sigma * 12 * pow(M_PI, 2) +
151  ChiM * kappa *
152  (std::sin(2 * M_PI * p[0] + M_PI / 4) *
153  std::sin(2 * M_PI * p[1] + M_PI / 4) *
154  std::sin(2 * M_PI * p[2] + M_PI / 4) *
155  std::exp(-5 * this->get_time()) -
156  a) *
157  (std::sin(2 * M_PI * p[0] + M_PI / 4) *
158  std::sin(2 * M_PI * p[1] + M_PI / 4) *
159  std::sin(2 * M_PI * p[2] + M_PI / 4) *
160  std::exp(-5 * this->get_time()) -
161  1) +
162  ChiM * (epsilon / (epsilon * gamma - 5)));
163  }
164  };

169  class BCNeumann : public lifex::Function<lifex::dim>

183  virtual double
184  value(const dealii::Point<lifex::dim> &p,
185  const unsigned int /*component*/ = 0) const override
186  {
187  if (lifex::dim == 2)
188  return Sigma *
189  (-2 * M_PI * std::sin(2 * M_PI * p[0]) *
190  std::cos(2 * M_PI * p[1]) *
191  std::exp(-5 * this->get_time()) * (std::abs(p[1]) < 1e-10) +
192  2 * M_PI * std::cos(2 * M_PI * p[0]) *
193  std::sin(2 * M_PI * p[1]) *
194  std::exp(-5 * this->get_time()) *
195  (std::abs(p[0] - 1) < 1e-10) +
196  2 * M_PI * std::sin(2 * M_PI * p[0]) *
197  std::cos(2 * M_PI * p[1]) *
198  std::exp(-5 * this->get_time()) *
199  (std::abs(p[1] - 1) < 1e-10) -
200  2 * M_PI * std::cos(2 * M_PI * p[0]) *
201  std::sin(2 * M_PI * p[1]) *
202  std::exp(-5 * this->get_time()) * (std::abs(p[0]) < 1e-10));
203  else
204  return Sigma *
205  (2 * M_PI * std::cos(2 * M_PI * p[0] + M_PI / 4) *
206  std::sin(2 * M_PI * p[1] + M_PI / 4) *
207  std::sin(2 * M_PI * p[2] + M_PI / 4) *
208  std::exp(-5 * this->get_time()) *
209  (std::abs(p[0] - 1) < 1e-10) -
210  2 * M_PI * std::cos(2 * M_PI * p[0] + M_PI / 4) *
211  std::sin(2 * M_PI * p[1] + M_PI / 4) *
212  std::sin(2 * M_PI * p[2] + M_PI / 4) *
213  std::exp(-5 * this->get_time()) * (std::abs(p[0]) < 1e-10) +
214  2 * M_PI * std::sin(2 * M_PI * p[0] + M_PI / 4) *
215  std::cos(2 * M_PI * p[1] + M_PI / 4) *
216  std::sin(2 * M_PI * p[2] + M_PI / 4) *
217  std::exp(-5 * this->get_time()) *
218  (std::abs(p[1] - 1) < 1e-10) -
219  2 * M_PI * std::sin(2 * M_PI * p[0] + M_PI / 4) *
220  std::cos(2 * M_PI * p[1] + M_PI / 4) *
221  std::sin(2 * M_PI * p[2] + M_PI / 4) *
222  std::exp(-5 * this->get_time()) * (std::abs(p[1]) < 1e-10) +
223  2 * M_PI * std::sin(2 * M_PI * p[0] + M_PI / 4) *
224  std::sin(2 * M_PI * p[1] + M_PI / 4) *
225  std::cos(2 * M_PI * p[2] + M_PI / 4) *
226  std::exp(-5 * this->get_time()) *

228  2 * M_PI * std::sin(2 * M_PI * p[0] + M_PI / 4) *
229  std::sin(2 * M_PI * p[1] + M_PI / 4) *
230  std::cos(2 * M_PI * p[2] + M_PI / 4) *
231  std::exp(-5 * this->get_time()) * (std::abs(p[2]) < 1e-10));

585  template <class basis>

587  MonodomainFHNDG<basis>::parse_parameters(lifex::ParamHandler &params)

594 
595  // Extra parameters.
596  params.enter_subsection("Mesh and space discretization");
597  this->prm_n_refinements = params.get_integer("Number of refinements");
598  this->prm_fe_degree = params.get_integer("FE space degree");
599  params.leave_subsection();
600 
601  params.enter_subsection("Discontinuous Galerkin");
602  this->prm_penalty_coeff = params.get_double("Penalty coefficient");
603  AssertThrow(this->prm_penalty_coeff == 1. || this->prm_penalty_coeff == 0. ||
604  this->prm_penalty_coeff == -1.,
605  dealii::StandardExceptions::ExcMessage(
606  "Penalty coefficient must be 1 (SIP method) or 0 (IIP method) "
607  "or -1 (NIP method)."));
608 
609  this->prm_stability_coeff = params.get_double("Stability coefficient");
610  params.leave_subsection();
611 
612  params.enter_subsection("Time solver");
613  this->prm_time_init = params.get_double("Initial time");
614  this->prm_time_final = params.get_double("Final time");
615  AssertThrow(this->prm_time_final > this->prm_time_init,
616  dealii::StandardExceptions::ExcMessage(
617  "Final time must be greater than initial time."));
618 
619  this->prm_time_step = params.get_double("Time step");
620  this->prm_bdf_order = params.get_integer("BDF order");
621  params.leave_subsection();
622 
623  params.enter_subsection("Parameters of the model");
624  ChiM = params.get_double("ChiM");
625  Cm = params.get_double("Cm");
626  Sigma = params.get_double("Sigma");
627  kappa = params.get_double("kappa");
628  epsilon = params.get_double("epsilon");
629  gamma = params.get_double("gamma");
630  a = params.get_double("a");
631  params.leave_subsection();
632  }

634  template <class basis>
635  void
636  MonodomainFHNDG<basis>::run()
637  {
638  this->create_mesh();
639  this->setup_system();
640 
641  this->initialize_solution(this->solution_owned, this->solution);
642  this->initialize_solution(this->solution_ex_owned, this->solution_ex);
643  this->initialize_solution(this->solution_owned_w, this->solution_w);
644  this->initialize_solution(this->solution_ex_owned_w, this->solution_ex_w);
645 
646  time_initialization();
647 
648  while (this->time < this->prm_time_final)
649  {
650  this->time += this->prm_time_step;
651  ++this->timestep_number;
652 
653  pcout << "Time step " << std::setw(6) << this->timestep_number
654  << " at t = " << std::setw(8) << std::fixed
655  << std::setprecision(6) << this->time << std::endl;
656 
657  this->update_time();
658  this->solution_ext = this->bdf_handler.get_sol_extrapolation();
659 
660  this->assemble_system();
661 
662  // Initial guess.
663  this->solution_owned = this->solution_ext;
664  this->solve_system();
665 
666  this->intermediate_error_print(this->solution_owned,
667  this->solution_ex_owned,
668  this->u_ex,
669  "u");
670  this->intermediate_error_print(this->solution_owned_w,
671  this->solution_ex_owned_w,
672  this->w_ex,
673  "w");
674  }
675 
676 
677  this->compute_errors(this->solution_owned,
678  this->solution_ex_owned,
679  this->u_ex,
680  this->grad_u_ex,
681  "u");
682  this->compute_errors(this->solution_owned_w,
683  this->solution_ex_owned_w,
684  this->w_ex,
685  this->grad_w_ex,
686  "w");
687 
688  // Generation of the graphical output.
689  this->solution_ex = this->solution_ex_owned;
690  this->conversion_to_fem(this->solution_ex);
691  this->solution = this->solution_owned;
692  this->conversion_to_fem(this->solution);
693  this->output_results();
694  }

718  template <class basis>
719  void
720  MonodomainFHNDG<basis>::time_initialization()
721  {
722  this->u_ex = std::make_shared<monodomain_fhn_DG::ExactSolution>();
723  this->grad_u_ex =
724  std::make_shared<monodomain_fhn_DG::GradExactSolution>();
725  this->f_ex = std::make_shared<monodomain_fhn_DG::RightHandSide>(
726  ChiM, Sigma, Cm, kappa, epsilon, gamma, a);
727  this->g_n = std::make_shared<monodomain_fhn_DG::BCNeumann>(Sigma);
728  w_ex =
729  std::make_shared<monodomain_fhn_DG::ExactSolution_w>(epsilon, gamma);

755  template <class basis>
756  void
757  MonodomainFHNDG<basis>::assemble_system()
758  {
759  const double &alpha_bdf = this->bdf_handler.get_alpha();
760 
761  solution_owned_w *= -gamma;
762  solution_owned_w.add(1, this->solution_owned);
763  solution_owned_w *= epsilon;
764  solution_owned_w.add(1 / this->prm_time_step, this->solution_bdf_w);
765  solution_owned_w *= this->prm_time_step / alpha_bdf;
766 
767  solution_w = solution_owned_w;
768 
769  this->matrix = 0;
770  this->rhs = 0;
771 
772  // The method is needed to define how the system matrix and rhs term are
773  // defined for the monodomain problem with Fithugh-Nagumo ionic model. The
774  // full matrix is composed by different sub-matrices that are called with
775  // simple capital letters. We refer here to the DG_Assemble methods for
776  // their definition.
777 
778  // See DG_Assemble::local_V().
779  dealii::FullMatrix<double> V(this->dofs_per_cell, this->dofs_per_cell);
780  // See DG_Assemble::local_M().
781  dealii::FullMatrix<double> M(this->dofs_per_cell, this->dofs_per_cell);
782  // See DG_Assemble::local_SC().
783  dealii::FullMatrix<double> SC(this->dofs_per_cell, this->dofs_per_cell);
784  // See DG_Assemble::local_IC().
785  dealii::FullMatrix<double> IC(this->dofs_per_cell, this->dofs_per_cell);
786  // Transpose of IC.
787  dealii::FullMatrix<double> IC_t(this->dofs_per_cell, this->dofs_per_cell);
788  // See DG_Assemble::local_IN().
789  dealii::FullMatrix<double> IN(this->dofs_per_cell, this->dofs_per_cell);
790  // Transpose of IN.
791  dealii::FullMatrix<double> IN_t(this->dofs_per_cell, this->dofs_per_cell);
792  // See DG_Assemble::local_SN().
793  dealii::FullMatrix<double> SN(this->dofs_per_cell, this->dofs_per_cell);
794  // See DG_Assemble::local_non_linear_fitzhugh().
795  dealii::FullMatrix<double> C(this->dofs_per_cell, this->dofs_per_cell);
796 
797  dealii::Vector<double> cell_rhs(this->dofs_per_cell);
798  dealii::Vector<double> cell_rhs_edge(this->dofs_per_cell);
799  dealii::Vector<double> u0_rhs(this->dofs_per_cell);
800  dealii::Vector<double> w0_rhs(this->dofs_per_cell);
801  std::vector<lifex::types::global_dof_index> dof_indices(
802  this->dofs_per_cell);
803 
804  dealii::IndexSet owned_dofs = this->dof_handler.locally_owned_dofs();
805  ;
806 
807  for (const auto &cell : this->dof_handler.active_cell_iterators())
808  {
809  if (cell->is_locally_owned())
810  {
811  this->assemble->reinit(cell);
812  dof_indices = this->dof_handler.get_dof_indices(cell);
813 
814  V = this->assemble->local_V();
815  V *= Sigma;
816  M = this->assemble->local_M();
817  M /= this->prm_time_step;
818  M *= alpha_bdf;
819  M *= ChiM;
820  M *= Cm;
821  C = this->assemble->local_non_linear_fitzhugh(this->solution_owned,
822  a,
823  dof_indices);
824  C *= kappa;
825  C *= ChiM;
826 
827  cell_rhs = this->assemble->local_rhs(this->f_ex);
828 
829  u0_rhs =
830  this->assemble->local_u0_M_rhs(this->solution_bdf, dof_indices);
831  u0_rhs /= this->prm_time_step;
832  u0_rhs *= ChiM;
833  u0_rhs *= Cm;
834 
835  w0_rhs =
836  this->assemble->local_w0_M_rhs(solution_owned_w, dof_indices);
837  w0_rhs *= ChiM;
838  w0_rhs *= (-1);
839 
840  this->matrix.add(dof_indices, V);
841  this->matrix.add(dof_indices, M);
842  this->matrix.add(dof_indices, C);
843  this->rhs.add(dof_indices, cell_rhs);
844  this->rhs.add(dof_indices, u0_rhs);
845  this->rhs.add(dof_indices, w0_rhs);
846 
847  for (const auto &edge : cell->face_indices())
848  {
849  this->assemble->reinit(cell, edge);
850  std::vector<lifex::types::global_dof_index> dof_indices_neigh(
851  this->dofs_per_cell);
852 
853  if (!cell->at_boundary(edge))
854  {
855  SC = this->assemble->local_SC(this->prm_stability_coeff);
856  SC *= Sigma;
857  std::tie(IC, IC_t) =
858  this->assemble->local_IC(this->prm_penalty_coeff);
859  IC *= Sigma;
860  IC_t *= Sigma;
861  this->matrix.add(dof_indices, SC);
862  this->matrix.add(dof_indices, IC);
863  this->matrix.add(dof_indices, IC_t);
864 
865  const auto neighcell = cell->neighbor(edge);
866  dof_indices_neigh =
867  this->dof_handler.get_dof_indices(neighcell);
868 
869  std::tie(IN, IN_t) =
870  this->assemble->local_IN(this->prm_penalty_coeff);
871  IN *= Sigma;
872  IN_t *= Sigma;
873  SN = this->assemble->local_SN(this->prm_stability_coeff);
874  SN *= Sigma;
875 
876  this->matrix.add(dof_indices, dof_indices_neigh, IN);
877  this->matrix.add(dof_indices_neigh, dof_indices, IN_t);
878  this->matrix.add(dof_indices, dof_indices_neigh, SN);
879  }
880  else
881  {
882  cell_rhs_edge =
883  this->assemble->local_rhs_edge_neumann(this->g_n);
884  this->rhs.add(dof_indices, cell_rhs_edge);
885  }
886  }
887  }
888  }
889 
890  this->matrix.compress(lifex::VectorOperation::add);
891  this->rhs.compress(lifex::VectorOperation::add);
892  }
